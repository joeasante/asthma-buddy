  <action>
    Create `app/controllers/settings/dose_logs_controller.rb`:

    ```ruby
    # frozen_string_literal: true

    module Settings
      class DoseLogsController < ApplicationController
        before_action :set_medication
        before_action :set_dose_log, only: :destroy

        def create
          @dose_log = @medication.dose_logs.new(dose_log_params)
          @dose_log.user = Current.user
          if @dose_log.save
            respond_to do |format|
              format.turbo_stream
              format.html { redirect_to settings_medications_path, notice: "Dose logged." }
            end
          else
            respond_to do |format|
              format.turbo_stream { render turbo_stream: turbo_stream.replace("dose_log_form_#{dom_id(@medication)}", partial: "settings/dose_logs/form", locals: { medication: @medication, dose_log: @dose_log }), status: :unprocessable_entity }
              format.html { redirect_to settings_medications_path, alert: "Could not log dose." }
            end
          end
        end

        def destroy
          @dose_log.destroy
          respond_to do |format|
            format.turbo_stream
            format.html { redirect_to settings_medications_path, notice: "Dose removed." }
          end
        end

        private

        def set_medication
          @medication = Current.user.medications.find(params[:medication_id])
        end

        def set_dose_log
          @dose_log = @medication.dose_logs.find(params[:id])
        end

        def dose_log_params
          params.require(:dose_log).permit(:puffs, :recorded_at)
        end
      end
    end
    ```

    Key decisions:
    - `set_medication` uses `Current.user.medications.find` — raises RecordNotFound (404) if medication belongs to another user (same pattern as MedicationsController).
    - `set_dose_log` uses `@medication.dose_logs.find` — raises RecordNotFound (404) if the dose log belongs to a different medication (and thus transitively to a different user).
    - `@dose_log.user = Current.user` set explicitly after `@medication.dose_logs.new` so both associations are populated before save.
    - Strong params permit only `:puffs` and `:recorded_at` — never `:user_id` or `:medication_id` from the form (sourced from URL and Current.user respectively).
    - Turbo Stream format responds for both success paths — views will be created in Plan 12-02.
    - Do NOT use `Current.user.dose_logs.new(medication: @medication)` — use `@medication.dose_logs.new` to keep association-scoped build consistent with Rails conventions for nested resources.

    Run `bin/rubocop app/controllers/settings/dose_logs_controller.rb` — must return 0 offenses.
