```ruby
test "user can edit an existing symptom entry inline" do

    assert_selector "form"

    click_button "Update symptom"
  end

  # After save, the entry in the list shows updated values

    assert_text "Chest tightness"
    assert_text "Severe"
    assert_no_selector "form"
  end
end
```

**Delete flow test:**

```ruby
# --- DELETE FLOW ---

test "user can delete a symptom entry and it disappears from the list" do
  entry = symptom_logs(:alice_wheezing)

  sign_in_as @alice
  visit symptom_logs_url

  assert_selector "##{dom_id(entry)}"

  # Dismiss the browser confirm dialog (accept it)
  # Capybara handles data-turbo-confirm via the browser's native dialog.
  # accept_confirm wraps the action that triggers the dialog.
  accept_confirm "Delete this entry?" do
    within("##{dom_id(entry)}") do
      click_button "Delete"
    end
  end

  # Entry no longer in the DOM
  assert_no_selector "##{dom_id(entry)}"
end
```

Notes:
- `accept_confirm "Delete this entry?"` — Capybara's `accept_confirm` matches the dialog message and accepts it. If the exact message text needs to match, use the string from `data-turbo-confirm` set in Plan 01.
- `within("##{dom_id(entry)}")` scopes interactions to the correct turbo frame element
- `click_link "Edit"` not `click_button` — the Edit control is a link (`link_to`) per Plan 01
- `click_button "Delete"` — the Delete control is a button rendered by `button_to` per Plan 01
- System tests use headless Chrome at 1400×1400 (established pattern, no change needed)
- `ActiveJob::Base.queue_adapter = :inline` is already in setup — no change needed
- If `accept_confirm` is not available in the app's system test helper, use `page.driver.browser.switch_to.alert.accept` as fallback, but `accept_confirm` is the Capybara 3.x standard API
