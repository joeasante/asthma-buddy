Create `app/controllers/accounts_controller.rb`:

```ruby
# frozen_string_literal: true

class AccountsController < ApplicationController
  def destroy
    if params[:confirmation] == "DELETE"
      Current.user.destroy
      reset_session
      redirect_to root_path, notice: "Your account and all associated data have been permanently deleted."
    else
      redirect_to settings_path, alert: "Account not deleted. You must type DELETE exactly to confirm."
    end
  end
end
```


**settings/show.html.erb — add Danger Zone section at the bottom:**

The settings page currently shows a nav grid with a Profile card. Append a Danger Zone section after the `.settings-nav-grid` div. Do NOT use a `<dialog>` confirm modal here — the typed confirmation field IS the confirmation mechanism. The form uses `data-turbo="false"` to avoid Turbo interfering with the session reset redirect.

```erb
<section class="danger-zone" aria-labelledby="danger-zone-heading">
  <h2 id="danger-zone-heading" class="danger-zone-heading">Danger Zone</h2>
  <div class="section-card section-card--danger">
    <div class="danger-zone-body">
      <div class="danger-zone-description">
        <h3 class="danger-zone-action-title">Delete account</h3>
        <p class="danger-zone-action-description">
          Permanently delete your account and all health data — symptom logs, peak flow readings,
          medications, dose logs, and health events. <strong>This cannot be undone.</strong>
        </p>
      </div>
      <%= form_with url: account_path, method: :delete, data: { turbo: false } do |f| %>
        <div class="field">
          <%= f.label :confirmation, 'Type "DELETE" to confirm', class: "field-label" %>
          <%= f.text_field :confirmation,
                class: "field-input field-input--danger",
                placeholder: "DELETE",
                autocomplete: "off",
                spellcheck: "false" %>
        </div>
        <%= f.submit "Permanently delete my account",
              class: "btn-danger",
              data: { confirm: false } %>
      <% end %>
    </div>
  </div>
</section>
```

Note: `data: { confirm: false }` prevents the global Stimulus confirm controller from intercepting this button — the typed field is the confirmation.

**test/controllers/accounts_controller_test.rb — create with these tests:**

```ruby
# frozen_string_literal: true

require "test_helper"

class AccountsControllerTest < ActionDispatch::IntegrationTest
  setup do
    @user = users(:verified_user)
    sign_in_as @user
  end

  test "DELETE /account with correct confirmation destroys user and redirects to root" do
    assert_difference "User.count", -1 do
      delete account_path, params: { confirmation: "DELETE" }
    end
    assert_redirected_to root_path
    assert_equal "Your account and all associated data have been permanently deleted.", flash[:notice]
  end

  test "DELETE /account with wrong confirmation does not destroy user" do
    assert_no_difference "User.count" do
      delete account_path, params: { confirmation: "delete" }
    end
    assert_redirected_to settings_path
    assert_equal "Account not deleted. You must type DELETE exactly to confirm.", flash[:alert]
  end

  test "DELETE /account with empty confirmation does not destroy user" do
    assert_no_difference "User.count" do
      delete account_path, params: { confirmation: "" }
    end
    assert_redirected_to settings_path
  end

  test "unauthenticated DELETE /account redirects to sign in" do
    sign_out
    delete account_path, params: { confirmation: "DELETE" }
    assert_redirected_to new_session_path
  end

  test "deleted user cannot sign back in" do
    delete account_path, params: { confirmation: "DELETE" }
    sign_out
    post session_path, params: { email_address: @user.email_address, password: "password123" }
    assert_response :unprocessable_entity
  end
end
```
