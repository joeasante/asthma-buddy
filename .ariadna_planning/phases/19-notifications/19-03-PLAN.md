```ruby
# frozen_string_literal: true

require "test_helper"

class NotificationsControllerTest < ActionDispatch::IntegrationTest
  setup do
    @user = users(:verified_user)
    sign_in_as @user
    @alice_unread = notifications(:alice_low_stock)
    @alice_read   = notifications(:alice_read_old)
    @bob_notif    = notifications(:bob_notification)
  end

  # ---------------------------------------------------------------
  # GET /notifications (index)
  # ---------------------------------------------------------------

  test "index returns 200 for authenticated user" do
    get notifications_path
    assert_response :success
  end

  test "index lists only current user's notifications" do
    get notifications_path
    assert_response :success
    # Alice's notification body appears
    assert_select ".notification-body", text: /#{Regexp.escape(@alice_unread.body)}/
    # Bob's notification body does NOT appear
    assert_no_match @bob_notif.body, response.body
  end

  test "index redirects unauthenticated users to sign in" do
    sign_out
    get notifications_path
    assert_redirected_to new_session_path
  end

  # ---------------------------------------------------------------
  # PATCH /notifications/:id/mark_read
  # ---------------------------------------------------------------

  test "mark_read marks notification as read" do
    assert_not @alice_unread.read
    patch mark_read_notification_path(@alice_unread),
          headers: { "Accept" => "text/vnd.turbo-stream.html" }
    assert_response :success
    assert @alice_unread.reload.read
  end

  test "mark_read returns Turbo Stream response" do
    patch mark_read_notification_path(@alice_unread),
          headers: { "Accept" => "text/vnd.turbo-stream.html" }
    assert_response :success
    assert_equal "text/vnd.turbo-stream.html", response.media_type
  end

  test "mark_read returns 404 for another user's notification" do
    patch mark_read_notification_path(@bob_notif),
          headers: { "Accept" => "text/vnd.turbo-stream.html" }
    assert_response :not_found
  end

  test "mark_read redirects unauthenticated users" do
    sign_out
    patch mark_read_notification_path(@alice_unread)
    assert_redirected_to new_session_path
  end

  # ---------------------------------------------------------------
  # POST /notifications/mark_all_read
  # ---------------------------------------------------------------

  test "mark_all_read marks all unread notifications as read" do
    unread_before = @user.notifications.unread.count
    assert unread_before > 0

    post mark_all_read_notifications_path,
         headers: { "Accept" => "text/vnd.turbo-stream.html" }
    assert_response :success

    assert_equal 0, @user.notifications.reload.unread.count
  end

  test "mark_all_read returns Turbo Stream response" do
    post mark_all_read_notifications_path,
         headers: { "Accept" => "text/vnd.turbo-stream.html" }
    assert_response :success
    assert_equal "text/vnd.turbo-stream.html", response.media_type
  end

  test "mark_all_read does not affect other users' notifications" do
    bob_unread_before = @bob_notif.reload.read
    post mark_all_read_notifications_path,
         headers: { "Accept" => "text/vnd.turbo-stream.html" }
    assert_equal bob_unread_before, @bob_notif.reload.read
  end

  test "mark_all_read redirects unauthenticated users" do
    sign_out
    post mark_all_read_notifications_path
    assert_redirected_to new_session_path
  end
end
```

Note: `sign_out` helper follows the pattern from existing controller tests — call `delete session_path` or use the SessionTestHelper sign_out method. Check test_helper.rb / session_test_helper.rb for the actual method name used in this codebase.

```ruby
# frozen_string_literal: true

require "application_system_test_case"

class NotificationsTest < ApplicationSystemTestCase
  setup do
    @user = users(:verified_user)
    sign_in_as @user

    # Create unread notifications inline for system test isolation
    # (do not rely solely on fixtures which may be shared)
    @low_stock_notif = Notification.create!(
      user:              @user,
      notification_type: :low_stock,
      notifiable:        medications(:alice_preventer),
      body:              "Clenil Modulite is running low.",
      read:              false
    )

    @missed_dose_notif = Notification.create!(
      user:              @user,
      notification_type: :missed_dose,
      notifiable:        medications(:alice_preventer),
      body:              "You haven't logged your Clenil Modulite dose today.",
      read:              false
    )
  end

  # ---------------------------------------------------------------
  # Badge visibility
  # ---------------------------------------------------------------

  test "unread badge appears on nav bell when unread notifications exist" do
    visit dashboard_path
    # Bell link should have data-unread-count > 0 on desktop nav or bottom nav
    assert_selector "[data-unread-count]:not([data-unread-count='0'])", wait: 5
  end

  test "nav bell badge disappears after all notifications marked read" do
    visit notifications_path
    click_button "Mark all read"
    # Badge should disappear (data-unread-count becomes 0 or element removed)
    assert_no_selector "[data-unread-count]:not([data-unread-count='0'])", wait: 5
  end

  # ---------------------------------------------------------------
  # Mark single notification as read
  # ---------------------------------------------------------------

  test "mark single notification as read updates row inline without full page reload" do
    visit notifications_path

    # Unread notification body is visible with bold styling
    assert_selector ".notification-body--unread", text: /Clenil Modulite is running low/, wait: 5

    # Click mark read on the first unread notification
    within(dom_id_selector(@low_stock_notif, :turbo_frame)) do
      click_button "Mark read"
    end

    # The row should update in place — unread dot gone, body no longer bold
    assert_no_selector "##{dom_id(@low_stock_notif)} .notification-unread-dot", wait: 5
    assert_no_selector "##{dom_id(@low_stock_notif)} .notification-body--unread", wait: 5
  end

  # ---------------------------------------------------------------
  # Mark all read
  # ---------------------------------------------------------------

  test "mark all read updates all rows and removes unread indicators" do
    visit notifications_path

    # Both notifications are unread initially
    assert_selector ".notification-row--unread", minimum: 2, wait: 5

    click_button "Mark all read"

    # All rows should now be read-styled
    assert_no_selector ".notification-row--unread", wait: 5
    assert_no_selector ".notification-unread-dot", wait: 5
  end

  # ---------------------------------------------------------------
  # Empty state
  # ---------------------------------------------------------------

  test "empty state 'You're all caught up.' shown when all notifications read" do
    # Mark all notifications as read
    @user.notifications.update_all(read: true)

    visit notifications_path

    assert_text "You're all caught up.", wait: 5
  end

  test "empty state shown when no notifications exist at all" do
    @user.notifications.destroy_all

    visit notifications_path

    assert_text "You're all caught up.", wait: 5
  end
end
```

**Helper method for turbo frame selector:**

The `dom_id_selector` helper is not a Rails built-in. Use inline selectors instead:
```ruby
within("##{ActionView::RecordIdentifier.dom_id(@low_stock_notif)}") do
  click_button "Mark read"
end
```

Or include `ActionView::RecordIdentifier` in the test and call `dom_id(@low_stock_notif)` directly (it's already available in system tests via ApplicationSystemTestCase which includes helpers).

**Capybara caveats from STATE.md to apply:**
- Use `wait: 5` or `wait: 10` for assertions after Turbo Stream updates — default 2s may be insufficient
- CSS `text-transform` can affect assert_text — use `assert_text /pattern/i` for case-insensitive matching where needed
- Use `assert_no_selector` (not assert_no_text) for CSS-class-based assertions
- `destroy_all` in setup/teardown is safe within Minitest transactions
