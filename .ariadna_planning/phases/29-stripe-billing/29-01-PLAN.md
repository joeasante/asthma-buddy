    1. Create `app/models/concerns/plan_limits.rb`:
       ```ruby
       module PlanLimits
         extend ActiveSupport::Concern

         def premium?
           admin? || payment_processor&.subscription&.active?
         end

         def free?
           !premium?
         end

         def plan_name
           premium? ? "Premium" : "Free"
         end

         def plan_features
           premium? ? PLANS[:premium][:features] : PLANS[:free][:features]
         end

         def subscription_status
           sub = payment_processor&.subscription
           return "none" unless sub
           if sub.active? && sub.ends_at.present?
             "cancelling"
           elsif sub.active?
             "active"
           else
             sub.status
           end
         end

         def next_billing_date
           payment_processor&.subscription&.current_period_end
         end
       end
       ```

    2. Update `app/models/user.rb`:

       - Add `pay_customer default_payment_processor: :stripe` after the alias
       - Place these three lines near the top of the class, after the existing includes
